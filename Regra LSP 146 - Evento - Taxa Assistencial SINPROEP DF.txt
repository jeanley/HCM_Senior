@ Definindo cursores, alfas e datas @
Definir Cursor cur_apf141;
Definir Cursor cur_seg141;
Definir Cursor cur_vlr141;
Definir Data dDatZer;
Definir Data dPerRef;
Definir Alfa aCodSeg;


@ Definindo e zerando variáveis @
nNumEmp = R034FUN.NumEmp;
nTipCol = R034FUN.TipCol;
nNumCad = R034FUN.NumCad;
dPerRef = FimCmp;
MontaData (31, 12, 1900, dDatZer);
nCodAti = 0;
aCodSeg = "";
nEnsInf = 0;
nEnsFun = 0;
nEnsMed = 0;
nValEIn = 0;
nValEFu = 0;
nValEMe = 0;

@ Se não for sócio do sindicato, zera o evento @
Se (aSocSin <> 'S')
  {
  RefEvt = 0;
  ValEvt = 0;
  Cancel(1);
  }


@ Cursor para buscar as atividades ativas no professor @
          cur_apf141.sql "select * from r171apf \
                           where numemp = :nNumEmp \
                             and tipcol = :nTipCol \
                             and numcad = :nNumCad \
                             and datini <= :dPerRef \
                             and (datfim is null \
                              or datfim = :dDatZer \
                              or datfim >= :dPerRef)";
          cur_apf141.AbrirCursor();
          Enquanto (cur_apf141.Achou)
            {
            nCodAti = cur_apf141.CodApf; 
            cur_seg141.sql "select USU_TipSeg from r171ati \ 
                              where CodApf = :nCodAti";
            cur_seg141.AbrirCursor();
              Se(cur_seg141.Achou)
                {
                 aCodSeg = cur_seg141.USU_TipSeg;
                  }
                 Se (aCodSeg = "EI")
                   {
                    nEnsInf = EnsInf + 1;
                   }
                 Senao
                 Se (aCodSeg = "EF")
                   {
                    nEnsFun = nEnsFun + 1;
                }
          
            nCodAti = 0;
            aCodSeg = "";
            cur_apf141.Proximo();
            }
          cur_apf141.FecharCursor();

cur_vlr141.sql "select * from r026vlr \
                  where comvlr <= :dPerRef \
                  and codvlr in (100, 101, 102) order by comvlr desc";

cur_vlr141.AbrirCursor();
  Enquanto (cur_vlr141.Achou)
    {
    nCodVlr = cur_vlr141.CodVlr;
    Se (nCodVlr = 200)
      {
      nValEIn = cur_vlr141.valvlr;
      }
    Senao
    Se (nCodVlr = 201)
      {
      nValEMe = cur_vlr141.valvlr;
      }
    
    nCodVlr = 0;
    cur_vlr141.Proximo();
    }

cur_vlr141.FecharCursor();                        

Se (nEnsMed > 0)
  {
   ValEvt = nValEMe;
   Cancel(1);
  }
Senao
Se (nEnsFun > 0)
  {
   ValEvt = nValEFu;
   Cancel(1);
  }
Senao
Se (nEnsInf > 0)
  {
   ValEvt = nValEIn;
   Cancel(1);
  }
Senao
Se ((nEnsFun = 0) e (nEnsInf = 0))
  {
  RefEvt = 0;
  ValEvt = 0;
  }
