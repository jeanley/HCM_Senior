select a.numemp,a.tipcol,a.numcad,a.codcal,a.oripen,a.tippen,sum(a.valpen) valpen,b.perref,b.datpag,
       c.codban,c.codage,c.tpctba,c.conban,c.digban,d.digage,c.ruadep,c.codcep,c.codcid,c.codbai,
       f.codfil,c.usu_nomfav,c.usu_cpffav
  from r036pjp a,
       r044cal b,
       r036pju c,
       r012age d,
       r036dep e,

 where a.numemp = b.numemp 
   and a.codcal = b.codcal
   and a.numemp = c.numemp 
   and a.tipcol = c.tipcol 
   and a.numcad = c.numcad 
   and a.coddep = c.coddep
   and c.codban = d.codban
   and c.codage = d.codage
   and a.numemp = e.numemp
   and a.tipcol = e.tipcol 
   and a.numcad = e.numcad 
   and a.coddep = e.coddep
   and a.numemp = f.numemp 
   and a.tipcol = f.tipcol 
   and a.numcad = f.numcad 
   and f.datalt = (select max(f2.datalt) 
                     from r038hfi f2
                    where f.numemp = f2.numemp
                      and f.tipcol = f2.tipcol 
                      and f.numcad = f2.numcad 
                      and f2.datalt <= eomonth(b.perref))
   and e.penjud = 'S'
   and c.modpag = 'R'
   and a.oripen = 'C'
