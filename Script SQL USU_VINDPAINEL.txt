select	a.numemp,
		a.tipcol,
		a.numcad,
		a.nomfun,
		a.postra,
		b.codmod,
		b.indcon,
		'update r034mdl set indcon = ''S'' where codmod = 17 and numemp = ' + cast(a.numemp as varchar) + ' and tipcol = ' + cast(a.tipcol as varchar) + ' and numcad = ' + cast(a.numcad as varchar) + ';' 
from	r034fun (nolock) a
		left join r034mdl b on b.numemp = a.numemp and b.tipcol = a.tipcol and b.numcad = a.numcad
where	a.tipcol in (1,2)
and		a.sitafa not in (7,18)
and		a.postra is not null
and		b.indcon <> 'S'
and		b.codmod = 17
and     a.postra not in('1', '1.000')
